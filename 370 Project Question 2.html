def decade(year):
    if 1920 < year.year < 1930:
        return '1920s'
    if 1930< year.year < 1940:
        return '1930s'
    if 1940< year.year < 1950:
        return '1940s'
    if 1950< year.year < 1960:
        return '1950s'
    if 1960< year.year < 1970:
        return '1960s'
    if 1970< year.year < 1980:
        return '1970s'
    if 1980< year.year < 1990:
        return '1980s'
    if 1990< year.year < 2000:
        return '1990s'
    if 2000< year.year < 2010:
        return '2000s'
    if 2010< year.year < 2020:
        return '2010s'

def year_num(date_str):
    date_format = '%Y'
    date_obj = datetime.strptime(date_str, date_format)
    return date_obj

def main_genre(genre):
    if ',' not in genre:
        return genre
    else:
        l = genre.split()
    return l[0][:-1]
In[]:
